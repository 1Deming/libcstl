2.2.30. deque_insert_range

  • 函数描述

向双端队列容器中插入一个数据区间。

  • 函数原型
void deque_insert_range(
    deque_t* pdeq_deque, deque_iterator_t it_pos, iterator_t it_begin, iterator_t it_end);
  • 参数说明
pdeq_deque: 双端队列容器。
it_pos: 插入数据的位置。
it_begin: 数据区间开始位置迭代器。
it_end: 数据区间末尾位置迭代器。
  • 备注

pdeq_deque必须有效的且已经初始化的双端队列容器，否则函数的行为是未定义的。 it_pos必须是属于pdeq_deque的有效的迭代器，否则函数的行为是未定义的。 [it_begin, it_end)必须是有效的数据区间，否则函数的行为是未定义的。 [it_begin, it_end)中的数据和pdeq_deque中的数据类型一致，否则函数的行为是未定义的。

  • 前提条件

包含头文件<cstl/cdeque.h>

  • 示例代码
/*
 * deque_insert_range.c
 */

#include <stdio.h>
#include <cstl/cdeque.h>
#include <cstl/clist.h>

int main(int argc, char* argv[])
{
    deque_t*            pdeq = NULL;
    list_t*             plist = NULL;
    deque_iterator_t    it;

    if ((pdeq = create_deque(int)) == NULL) {
        return -1;
    }
    if ((plist = create_list(int)) == NULL) {
        deque_destroy(pdeq);
        return -1;
    }

    deque_init(pdeq);
    deque_push_back(pdeq, 10);
    deque_push_back(pdeq, 20);
    deque_push_back(pdeq, 30);

    list_init(plist);
    list_push_back(plist, -10);
    list_push_back(plist, -20);
    list_push_back(plist, -30);

    printf("pdeq = ");
    for (it = deque_begin(pdeq);
         !iterator_equal(it, deque_end(pdeq));
         it = iterator_next(it)) {
        printf("%d ", *(int*)iterator_get_pointer(it));
    }
    printf("\n");

    deque_insert_range(pdeq, deque_begin(pdeq), list_begin(plist), list_end(plist));

    printf("pdeq = ");
    for (it = deque_begin(pdeq);
         !iterator_equal(it, deque_end(pdeq));
         it = iterator_next(it)) {
        printf("%d ", *(int*)iterator_get_pointer(it));
    }
    printf("\n");

    deque_destroy(pdeq);
    list_destroy(plist);

    return 0;
}
  • 结果输出

pdeq = 10 20 30
pdeq = -10 -20 -30 10 20 30
